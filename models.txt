### products

```python
from sqlalchemy import Column, Integer, String, Numeric, Boolean, ForeignKey
from app.database import Base

class Product(Base):
    __tablename__ = "products"

    id = Column(Integer, primary_key=True)
    category_id = Column(Integer, ForeignKey("categories.id", ondelete="CASCADE"))
    supplier_id = Column(Integer, ForeignKey("suppliers.id", ondelete="SET NULL"), nullable=True)

    barcode = Column(String(255), unique=True)
    name = Column(String(255))
    purchase_price = Column(Numeric(12, 2))
    sale_price = Column(Numeric(12, 2))
    alert_stock = Column(Integer, default=0)
    is_active = Column(Boolean, default=True)
```

### purchases

```python
from sqlalchemy import Column, Integer, String, Numeric, Enum, ForeignKey
from app.database import Base
import enum

class PurchaseStatus(enum.Enum):
    pending = "pending"
    partial = "partial"
    paid = "paid"

class Purchase(Base):
    __tablename__ = "purchases"

    id = Column(Integer, primary_key=True)
    supplier_id = Column(Integer, ForeignKey("suppliers.id", ondelete="CASCADE"))
    user_id = Column(Integer, ForeignKey("users.id", ondelete="CASCADE"))

    reference = Column(String(255), unique=True)
    total_amount = Column(Numeric(12, 2))
    paid_amount = Column(Numeric(12, 2), default=0)
    status = Column(Enum(PurchaseStatus), default=PurchaseStatus.pending)
```

### sales

```python
from sqlalchemy import Column, Integer, String, Numeric, Enum, ForeignKey
from app.database import Base
import enum

class SaleStatus(enum.Enum):
    paid = "paid"
    partial = "partial"
    credit = "credit"

class PaymentMethod(enum.Enum):
    cash = "cash"
    card = "card"
    mobile = "mobile"

class Sale(Base):
    __tablename__ = "sales"

    id = Column(Integer, primary_key=True)
    customer_id = Column(Integer, ForeignKey("customers.id", ondelete="SET NULL"), nullable=True)
    user_id = Column(Integer, ForeignKey("users.id", ondelete="CASCADE"))

    reference = Column(String(255), unique=True)
    total_amount = Column(Numeric(12, 2))
    discount = Column(Numeric(12, 2), default=0)
    paid_amount = Column(Numeric(12, 2), default=0)

    payment_method = Column(Enum(PaymentMethod), nullable=True)
    status = Column(Enum(SaleStatus), default=SaleStatus.paid)
```

### stock_movements (🔥 cœur du système)

```python
from sqlalchemy import Column, Integer, String, Enum, ForeignKey, Text, DateTime, Index
from app.database import Base
import enum
from datetime import datetime

class StockType(enum.Enum):
    in_ = "in"
    out = "out"
    adjust = "adjust"

class StockMovement(Base):
    __tablename__ = "stock_movements"

    id = Column(Integer, primary_key=True)
    product_id = Column(Integer, ForeignKey("products.id", ondelete="CASCADE"))
    user_id = Column(Integer, ForeignKey("users.id", ondelete="CASCADE"))

    type = Column(Enum(StockType))
    quantity = Column(Integer)

    source_type = Column(String(50))
    source_id = Column(Integer, nullable=True)

    note = Column(Text)
    created_at = Column(DateTime, default=datetime.utcnow)

    __table_args__ = (
        Index("idx_product_source", "product_id", "source_type"),
    )
```

### payments (polymorphique)

```python
from sqlalchemy import Column, Integer, String, Numeric, ForeignKey
from app.database import Base

class Payment(Base):
    __tablename__ = "payments"

    id = Column(Integer, primary_key=True)
    payable_type = Column(String(50))   # sale / purchase
    payable_id = Column(Integer)

    amount = Column(Numeric(12, 2))
    method = Column(String(50))

    user_id = Column(Integer, ForeignKey("users.id", ondelete="CASCADE"))
```

## 4️⃣ Création des tables (ordre important)

```python
from app.database import engine, Base
from app.models import *

Base.metadata.create_all(bind=engine)
```
