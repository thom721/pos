# 1. Faire confiance au certificat dans macOS
sudo security add-trusted-cert -d -r trustRoot \
  -k /Library/Keychains/System.keychain \
  certificat/server.crt

# 2. Ajouter le hostname
echo "127.0.0.1	infini-post.local" | sudo tee -a /etc/hosts

# 3. Démarrer les services
docker compose up -d

# 4. Tester
curl https://infini-post.local/api/public/config
