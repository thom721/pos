## 1️⃣ Règles métier (inchangées ✔️)

| Règle                     | Implémentation             |
| ------------------------- | -------------------------- |
| Stock insuffisant         | Exception → rollback       |
| Vente = stock out         | `StockMovement(out, -qty)` |
| Paiement = ligne payments | OK                         |
| Transaction DB            | `session.begin()`          |
| Pas de stock direct       | uniquement mouvements      |

```python
from pydantic import BaseModel
from typing import List, Optional
from uuid import UUID

class SaleItemInput(BaseModel):
    product_id: UUID
    quantity: int

class SaleCreate(BaseModel):
    customer_id: Optional[UUID] = None
    discount: float = 0
    paid_amount: float = 0
    payment_method: Optional[str] = None
    items: List[SaleItemInput]
```

## 3️⃣ Service de vente (🔥 cœur du POS)

### 📁 `app/services/sale_service.py`

```python
from sqlalchemy.orm import Session
from fastapi import HTTPException
from datetime import datetime

from app.models import (
    Sale, SaleItem, Product,
    StockMovement, Payment
)

def create_sale(
    db: Session,
    data,
    user_id: str
):
    with db.begin():  # 🔥 TRANSACTION ATOMIQUE

        if not data.items:
            raise HTTPException(400, "Aucun produit")

        total = 0

        # 1️⃣ Vérification stock + calcul total
        for item in data.items:
            product = db.get(Product, str(item.product_id))

            if not product:
                raise HTTPException(404, "Produit introuvable")

            if product.stock < item.quantity:
                raise HTTPException(
                    400,
                    f"Stock insuffisant pour {product.name}"
                )

            total += product.sale_price * item.quantity

        discount = data.discount or 0
        total_after_discount = total - discount

        # 2️⃣ Création vente
        sale = Sale(
            customer_id=str(data.customer_id) if data.customer_id else None,
            user_id=user_id,
            reference=f"VNT-{int(datetime.utcnow().timestamp())}",
            total_amount=total_after_discount,
            discount=discount,
            paid_amount=data.paid_amount,
            status="credit" if data.paid_amount < total_after_discount else "paid"
        )
        db.add(sale)
        db.flush()  # récupère sale.id

        # 3️⃣ Items + mouvements stock
        for item in data.items:
            product = db.get(Product, str(item.product_id))

            db.add(SaleItem(
                sale_id=sale.id,
                product_id=product.id,
                quantity=item.quantity,
                unit_price=product.sale_price,
                subtotal=product.sale_price * item.quantity
            ))

            db.add(StockMovement(
                product_id=product.id,
                user_id=user_id,
                type="out",
                quantity=-item.quantity,
                source_type="sale",
                source_id=sale.id,
                note="Vente POS"
            ))

        # 4️⃣ Paiement
        if data.paid_amount > 0:
            db.add(Payment(
                payable_type="sale",
                payable_id=sale.id,
                amount=data.paid_amount,
                method=data.payment_method,
                user_id=user_id
            ))

        return sale
```

## 4️⃣ Endpoint FastAPI

### 📁 `app/routes/sales.py`

```python
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session

from app.database import get_db
from app.schemas.sale import SaleCreate
from app.services.sale_service import create_sale

router = APIRouter(prefix="/sales", tags=["Sales"])

@router.post("/", status_code=201)
def store_sale(
    payload: SaleCreate,
    db: Session = Depends(get_db),
    user_id: str = "USER_UUID_Ici"  # remplacé par JWT plus tard
):
    sale = create_sale(db, payload, user_id)

    return {
        "message": "Vente enregistrée avec succès",
        "sale_id": sale.id
    }
```

## 5️⃣ Annulation / retour de vente

### 📁 `app/services/sale_service.py`

```python
def cancel_sale(db: Session, sale_id: str, user_id: str):
    with db.begin():

        sale = db.get(Sale, sale_id)

        if not sale:
            raise HTTPException(404, "Vente introuvable")

        for item in sale.items:
            db.add(StockMovement(
                product_id=item.product_id,
                user_id=user_id,
                type="in",
                quantity=item.quantity,
                source_type="sale_cancel",
                source_id=sale.id,
                note="Annulation vente"
            ))

        sale.status = "cancelled"
```
